import pandas as pd
import altair as alt
import sqlite3

# Grand Question 1

con = sqlite3.connect('lahmansbaseballdb.sqlite')

test = pd.read_sql_query("SELECT * FROM CollegePlaying", con)
test.loc[test.schoolID.str.contains('byui')]
test.query('schoolID == "idbyuid"').head()

q1 = pd.read_sql_query("""
    SELECT DISTINCT s.playerID, c.schoolID, s.salary, s.yearID, s.teamID
    FROM Salaries s
        JOIN CollegePlaying c
        ON c.playerID = s.playerID
    WHERE schoolID == 'idbyuid'
    ORDER BY s.salary DESC
    """, con)
# (s.salary/1000) as 'salary (k)'

print(q1)

# Grand Question 2a

q2a = pd.read_sql_query("""
    SELECT playerID, yearID, H, AB, (CAST(H AS float) / AB) AS batting_average
    FROM batting
    WHERE AB > 1
    ORDER BY batting_average DESC, playerID
    LIMIT 5
    """, con)

print(q2a)

# Grand Question 2b

q2b = pd.read_sql_query("""
    SELECT playerID, yearID, H, AB, (CAST(H AS float) / AB) AS batting_average
    FROM batting
    WHERE AB > 10
    ORDER BY batting_average DESC, playerID
    LIMIT 5
    """, con)

print(q2b)

# Grand Question 2c

q2c = pd.read_sql_query("""
    SELECT playerID, SUM(H) AS H, SUM(AB) as AB, (CAST(SUM(H) AS float) / SUM(AB)) AS batting_average
    FROM batting
    GROUP BY playerID
    HAVING SUM(AB) >= 100
    ORDER BY batting_average DESC
    LIMIT 5
    """, con)

print(q2c)
